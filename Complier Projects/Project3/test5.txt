TYPE
   at : INT;
   bt : at;
   ct : dt;
   et : BOOLEAN;
   ft : STRING;
   gt : REAL;

 VAR  
   av : at;   bv : bt;   cv : ct;   ev : et;   fv : ft;   gv : gt;
   hv : t1;   iv : t1;   jv : t2;   kv : t2;   lv : t3;

 {
    av = 1;
    bv = av + lv;
    WHILE jv < iv
    {
       av = av+1;
    }

    WHILE jv
    {
       av = av+1;
    }

    bv = av;
    cv = 1.0;
    mv = nv;
 }

INT : at bt t3 av bv lv #
REAL : ct gt dt cv gv #
STRING : ft fv #
BOOLEAN : et t1 t2 ev hv iv jv kv #
mv : nv #
